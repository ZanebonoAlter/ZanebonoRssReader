修改Start方法添加重试逻辑（替换第70-86行）：

```go
func (q *TagQueue) Start() error {
    q.mu.Lock()
    defer q.mu.Unlock()

    if q.started {
        return fmt.Errorf("tag queue already started")
    }

    maxRetries := 10
    retryInterval := 30 * time.Second

    for attempt := 1; attempt <= maxRetries; attempt++ {
        err := q.tryStart()
        if err == nil {
            if attempt > 1 {
                log.Printf("[INFO] TagQueue started after %d retry attempts", attempt-1)
            } else {
                log.Println("Tag queue started successfully")
            }
            return nil
        }

        log.Printf("[INFO] TagQueue retry attempt %d/%d: %v", attempt, maxRetries, err)

        if attempt < maxRetries {
            q.mu.Unlock()
            time.Sleep(retryInterval)
            q.mu.Lock()
        }
    }

    log.Printf("[ERROR] TagQueue failed to start after %d attempts, giving up", maxRetries)
    return fmt.Errorf("tag queue failed to start after %d attempts", maxRetries)
}

func (q *TagQueue) tryStart() error {
    // 验证数据库连接可用
    if sqlDB, err := database.DB.DB(); err != nil {
        return fmt.Errorf("failed to get database connection: %w", err)
    } else if err := sqlDB.Ping(); err != nil {
        return fmt.Errorf("database ping failed: %w", err)
    }

    // 尝试初始化队列（验证tag_jobs表可访问）
    queue := NewTagJobQueue(database.DB)
    if queue == nil {
        return fmt.Errorf("failed to create tag job queue")
    }

    q.stopChan = make(chan struct{})
    q.queue = queue
    q.started = true
    q.wg.Add(1)
    go q.worker()
    return nil
}
```

注意：
- retryInterval = 30 * time.Second (D-06)
- maxRetries = 10 (D-07, 约5分钟)
- 每次重试记录日志 [INFO] TagQueue retry attempt N/10 (D-08)
- 成功后记录日志 [INFO] TagQueue started after N retry attempts (D-09)
- tryStart() 验证数据库连接(ping)和队列初始化，检测真实启动失败而非空返回
