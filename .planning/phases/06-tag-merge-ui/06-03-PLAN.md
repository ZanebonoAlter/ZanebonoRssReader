   async function mergeSingle(candidate: TagMergeCandidate) {
     const customName = customNames.value.get(candidate.sourceTagId) || candidate.targetLabel
     mergingIds.value.push(candidate.sourceTagId)
     
     const result = await useTagMergePreviewApi().mergeTagsWithCustomName({
       sourceTagId: candidate.sourceTagId,
       targetTagId: candidate.targetTagId,
       newName: customName,
     })
     
     mergingIds.value = mergingIds.value.filter(id => id !== candidate.sourceTagId)
     
     if (result.success) {
       mergedIds.value.push(candidate.sourceTagId)
     } else {
       failedIds.value.push(candidate.sourceTagId)
       error.value = result.error || '合并失败'
     }
   }
