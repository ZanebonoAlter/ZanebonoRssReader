    ```go
    import (
        "log"

        "gorm.io/gorm"
        "my-robot-backend/internal/domain/models"
        "my-robot-backend/internal/platform/database"
    )

    type BlockedArticleRecoveryScheduler struct {
        interval      time.Duration
        stopChan      chan struct{}
        isRunning     bool
        isExecuting   bool
        mutex         sync.Mutex
    }

    func NewBlockedArticleRecoveryScheduler(intervalSeconds int) *BlockedArticleRecoveryScheduler {
        return &BlockedArticleRecoveryScheduler{
            interval:  time.Duration(intervalSeconds) * time.Second,
            stopChan:  make(chan struct{}),
            isRunning: false,
        }
    }
    ```

    2. **Start/Stop方法：** 参考auto_refresh.go的实现模式

    3. **核心恢复逻辑（runRecoveryCycle）：**
    ```go
    func (s *BlockedArticleRecoveryScheduler) runRecoveryCycle() {
        s.mutex.Lock()
        s.isExecuting = true
        s.mutex.Unlock()
        defer func() {
            s.mutex.Lock()
            s.isExecuting = false
            s.mutex.Unlock()
        }()

        // STAT-04: 恢复blocked文章（per D-04, D-05, D-06）
        // 查询所有firecrawl_status为waiting_for_firecrawl或blocked的文章
        var blockedArticles []models.Article
        err := database.DB.
            Joins("JOIN feeds ON feeds.id = articles.feed_id").
            Where("articles.firecrawl_status IN ?", []string{"waiting_for_firecrawl", "blocked"}).
            Find(&blockedArticles).Error
        
        if err != nil {
            log.Printf("[ERROR] BlockedArticleRecovery: failed to query blocked articles: %v", err)
            return
        }

        recoveredCount := 0
        for _, article := range blockedArticles {
            var feed models.Feed
            if err := database.DB.First(&feed, article.FeedID).Error; err != nil {
                // D-06: feed已删除则跳过（防御性判断）
                continue
            }

            // D-05: feed.FirecrawlEnabled变为true时，解除阻塞
            if feed.FirecrawlEnabled {
                result := database.DB.Model(&models.Article{}).
                    Where("id = ?", article.ID).
                    Update("firecrawl_status", "pending")
                
                if result.Error == nil && result.RowsAffected > 0 {
                    recoveredCount++
                    log.Printf("[INFO] BlockedArticleRecovery: recovered article %d from feed %d", article.ID, feed.ID)
                }
            }
        }

        if recoveredCount > 0 {
            log.Printf("[INFO] BlockedArticleRecovery: recovered %d blocked articles", recoveredCount)
        }

        // STAT-05: 阻塞数量警告（per D-07, D-08）
        var blockedCount int64
        err = database.DB.Model(&models.Article{}).
            Joins("JOIN feeds ON feeds.id = articles.feed_id").
            Where("articles.summary_status = ?", "incomplete").
            Where("feeds.article_summary_enabled = ?", true).
            Where("articles.firecrawl_status <> ?", "completed").
            Count(&blockedCount).Error
        
        if err != nil {
            log.Printf("[ERROR] BlockedArticleRecovery: failed to count blocked articles: %v", err)
            return
        }

        // D-08: 超过阈值50时输出WARN日志
        if blockedCount > 50 {
            log.Printf("[WARN] ContentCompletion blocked articles exceeded threshold: %d > 50", blockedCount)
        }
    }
    ```

    4. **定时循环：** 使用time.Ticker，参考auto_refresh.go模式

    - intervalSeconds参数设为3600（每小时一次，per D-04）
    - 阻塞阈值固定为50（per D-07）
    - 使用WARN级别日志输出警告（per D-08）
    - 复用现有GORM查询模式
